package com.mycompany.app.linkScaner.reader;

import com.mycompany.app.linkScaner.reader.ILinkReader;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;


import java.io.*;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.ArrayList;
import java.util.List;

public class LinkReader implements ILinkReader {

    private List<String> links;

    public LinkReader() {
        links = new ArrayList<>();

    }

    public void readLinksFromFile(String [] filesName) {
        for(String fileName : filesName) {
            Path path = Paths.get(fileName);
            File file = new File(path.toString());
            try {
                Document doc = Jsoup.parse(file, "UTF-8");
                Elements linkElements = doc.getElementsByTag("a");
                for(Element item : linkElements) {
                    links.add(item.attr("href"));
                    links.add(item.attr("src"));
                }
            } catch (IOException exception) {
                exception.printStackTrace();
            }
        }

    }

    public List<String> getLinks() {
        return links;
    }
}
